#include <iostream>
#include <string>
using namespace std;

int Rotate(bool turn, int step, int number) {
    if (turn) {
        number = number + step;
    }
    else {
        number = number - step;
    }

    number = number % 100;

    if (number < 0) {
        number = number + 100;
    }

    return number;
}

int main() {
    string input;
    getline(cin, input);

    int number = 50;
    bool LR = true;
    int step = 0;
    int totalZeroHits = 0;

    for (int i = 0; i <= input.length(); i++) {

        if (i < input.length() && input[i] == 'L') LR = false;
        if (i < input.length() && input[i] == 'R') LR = true;

        if (i < input.length() && isdigit(input[i])) {
            step = step * 10 + (input[i] - '0');
        }

        if (i == input.length() || input[i] == ' ') {
            if (step > 0) {

                number = Rotate(LR, step, number);

                if (number == 0)
                    totalZeroHits++;
            }
            step = 0;
        }
    }

    cout << totalZeroHits << endl;
}
