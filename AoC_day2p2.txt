#include <iostream>
#include <string>
#include <vector>
using namespace std;

bool isaValidNum(long long inputNum) {

    string stringt = to_string(inputNum);
    int len = stringt.size();

    for (int k = 1; k <= len / 2; k++) {
        if (len % k != 0) continue;

        int repeats = len / k;

        if (repeats < 2) { continue; }

        string block = stringt.substr(0, k);

        string test = "";
        for (int i = 0; i < repeats; i++)
            test += block;

        if (test == stringt) return true;
    }
    return false;
}

int main() {

    string line;
    getline(cin, line);

    vector<pair<long long, long long>> ranges;

    long long a = 0, b = 0;
    bool firsNum = true;

    for (char c : line) {
        if (isdigit(c)) {
            if (firsNum) a = a * 10 + (c - '0');

            else b = b * 10 + (c - '0');
        }
        else if (c == '-') {
            firsNum = false;
        }
        else if (c == ',') {
            ranges.emplace_back(a, b);
            a = b = 0;
            firsNum = true;
        }
    }

    ranges.emplace_back(a, b);

    long long total = 0;

    for (auto& pair : ranges) {
        for (long long n = pair.first; n <= pair.second; n++) {
            if (isaValidNum(n)) {
                total += n;
            }
        }
    }

    cout << total;
}
