#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    ifstream infile("input.txt");
    string line;
    long long total = 0;

    while (getline(infile, line)) {
        int n = line.size();
        int maxJoltage = 0;

        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int val = (line[i] - '0') * 10 + (line[j] - '0');
                maxJoltage = max(maxJoltage, val);
            }
        }

        total += maxJoltage;
    }

    cout << total << endl;
    return 0;
}
